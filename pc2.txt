#include <stdio.h>
#include <stdlib.h>
#include <omp.h>

int main(int argc,char **argv){
  int n = argc>1?atoi(argv[1]):8;
  omp_set_num_threads(argc>2?atoi(argv[2]):2);
  #pragma omp parallel for schedule(static,2)
  for(int i=0;i<n;i++){
    if(i%2==0){
      int e = (i+1<n)?i+1:n-1;
      printf("T%d: %d-%d\n", omp_get_thread_num(), i, e);
    }
  }
  return 0;
}
